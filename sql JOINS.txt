LEFT JOIN BOOKS
ON AUTHORS.AUTHORID = BOOKS.AID


		RIGHT OUTER JOIN
SELECT AUTHORNAME,CITY, BOOKNAME,BOOKPRICE FROM AUTHORS
RIGHT JOIN BOOKS
ON AUTHORS.AUTHORID = BOOKS.AID


		FULL OUTER JOIN

SELECT AUTHORNAME,CITY, BOOKNAME,BOOKPRICE FROM AUTHORS
FULL JOIN BOOKS
